23 def search(self, obj, exact=False, db=None): 

24 """ 

25 Search the database for partial matches of [obj], and return a list of matches 

26 in the tuple form: 

27 

28 ("obj", {"file_hash": string, "cryptographer": string, "storage_provider": string, "bucket": string}) 

29 

30 If [exact] == True, then only exact matches will be returned. Since there should 

31 only ever be a single exact match for a path in the DB, a CstashCriticalException 

32 will be thrown if more than a single element is in the resulting list. This shouldn't 

33 be possible anyway, since the DB is a key/value store, but it's a safety measure. 

34 """ 

35 

36 db = db or self.db 

37 db_connection = SqliteDict(db, autocommit=True, flag='r') 

38 if exact is True: 

39 keys = [(k, db_connection[k]) for k in db_connection.keys() if obj == k] 

40 else: 

41 keys = [(k, db_connection[k]) for k in db_connection.keys() if obj in k] 

42 

43 if exact is True and len(keys) > 1: 

44 raise exceptions.CstashCriticalException(message=(f"Found more than a single match " 

45 "for {obj} in the database:\n\n{keys}")) 

46 

47 db_connection.close() 

48 

49 return keys

51 def file_hash(self, filepath): 

52 """ 

53 Return the sha256 hash for [filepath], or False on failure 

54 """ 

55 

56 try: 

57 sha256 = hashlib.sha256() 

58 with open(filepath, 'rb') as f: 

59 for block in iter(lambda: f.read(4096), b''): 

60 sha256.update(block) 

61 return sha256.hexdigest() 

62 except FileNotFoundError as e: 

63 logging.error(f"Couldn't hash the file {filepath}: {e}") 

64 return False
